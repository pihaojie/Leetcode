题目：给定一个已排序的链表的头 head ， 删除所有重复的元素，使每个元素只出现一次 。返回 已排序的链表 。
代码：
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(!head)
           return 0;
        ListNode* cur=head;
        while(cur->next)
        {
            if(cur->val==cur->next->val)
            {
                cur->next=cur->next->next;
            }
            else
            {
                cur=cur->next;
            }
        }
        return head;
    }
};
