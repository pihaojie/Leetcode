题目：给你一个整数 n ，求恰由 n 个节点组成且节点值从 1 到 n 互不相同的 二叉搜索树有多少种？返回满足题意的二叉搜索树的种数。
思路：迭代，选定根节点，然后二叉树种数等于左子树种数乘右子树种数，对于整数n,先求第一个数为根节点的种数，再求第二个数为根节点的种数，以此类推。
代码：
class Solution {
public:
    int numTrees(int n) {
    vector<int> g(n+1,0);
    g[0]=1;
    g[1]=1;
    for(int i=2;i<=n;i++)
    {
        for(int j=1;j<=i;j++)
        {
            g[i]+=g[j-1]*g[i-j];
        }
    }
    return g[n];
    }
};
